// 3. 데이터 서비스 객체 (기존 LocalDB 대체)
const DB = {
  async fetchAll(table) {
    const { data, error } = await supabaseClient.from(table).select('*').schema('finance');
    if (error) { console.error(error); return []; }
    return data;
  },
  async insert(table, row) {
    const { data, error } = await supabaseClient.from(table).insert([row]).schema('finance').select();
    if (error) throw error;
    return data[0];
  }
};

// 4. 페이지 렌더링 로직 (비동기 처리)
async function renderDashboard() {
  const container = document.getElementById('db-content');
  try {
    const companies = await DB.fetchAll('card_companies');
    let html = `<div class="card"><h3>등록된 카드사 (${companies.length})</h3><ul>`;
    companies.forEach(c => {
      html += `<li>${c.name} (${c.payer})</li>`;
    });
    html += `</ul></div>`;
    container.innerHTML = html;
  } catch (e) {
    container.innerHTML = `<p style="color:red">데이터를 불러오지 못했습니다.</p>`;
  }
}

    if(n.dataset.page === 'dashboard') renderDashboard();
